/**
 * FUNCIÓN PRINCIPAL (El Director de Orquesta)
 * Coordina las fases: Extraer -> Transformar -> Cargar -> Mostrar Éxito
 */
function main() {
  Logger.log("--- INICIANDO PROCESO ETL ---");

  try {
    // 1. EXTRAER (E) - Ahora conecta directamente con el Data Lake de Drive
    const datosCrudos = extraerDatosCloud();
    if (!datosCrudos) return; // Si no hay datos, la función extraer ya lanzó el error

    // 2. TRANSFORMAR (T)
    const datosProcesados = transformarDatos(datosCrudos);

    // 3. CARGAR (L)
    if (datosProcesados && datosProcesados.length > 1) {
      
      cargarDatos(datosProcesados); 

      // 4. CONTROL DE CALIDAD Y MODAL
      Logger.log("Ejemplo de cliente filtrado: " + JSON.stringify(datosProcesados[1]));
      
      // Llamamos a la función que lanza la ventana HTML
      mostrarModalExito(datosProcesados.length);
      
      Logger.log("--- PROCESO COMPLETADO CON ÉXITO ---");
      
    } else {
      SpreadsheetApp.getUi().alert("No se encontraron clientes que cumplan el filtro (Churn = 1).");
    }

  } catch (error) {
    Logger.log("ERROR CRÍTICO EN MAIN: " + error.message);
    SpreadsheetApp.getUi().alert("Hubo un error en el proceso: " + error.message);
  }
}

/**
 * FASE E NUEVA: EXTRACCIÓN DESDE LA NUBE
 */
function extraerDatosCloud() {
  // 1. Buscamos el archivo en nuestro "Data Lake" (Google Drive)
  var nombreArchivo = "datos_gimnasio_cloud.csv";
  var iteradorArchivos = DriveApp.getFilesByName(nombreArchivo);
  
  // Verificamos si el archivo existe
  if (iteradorArchivos.hasNext()) {
    var archivo = iteradorArchivos.next();
    
    // 2. Extraemos el contenido en formato texto puro
    var contenidoTexto = archivo.getBlob().getDataAsString();
    
    // 3. ¡LA MAGIA! Convertimos el texto CSV en una matriz bidimensional
    var matrizDatos = Utilities.parseCsv(contenidoTexto);
    
    Logger.log("✅ Datos extraídos de la nube correctamente. Filas: " + matrizDatos.length);
    
    // Devolvemos la matriz. A partir de aquí, vuestro código de Calcular.gs 
    // funcionará EXACTAMENTE igual que antes, ¡porque la estructura es la misma!
    return matrizDatos;
    
  } else {
    Logger.log("❌ ERROR: No se encontró el archivo en Drive. Revisa que estés usando el mismo correo.");
    return null;
  }
}

function carreraVelocidadJS() {
  var datos = extraerDatosCloud(); // Trae las 50.000 filas
  var inicio = new Date().getTime();
  
  var sumaEdad = 0;
  var totalFilas = datos.length;
  
  // Recorremos las 50.000 filas con un bucle
  for (var i = 1; i < totalFilas; i++) {
    sumaEdad += Number(datos[i][0]); // La edad está en la primera columna
  }
  
  var promedio = sumaEdad / (totalFilas - 1);
  var fin = new Date().getTime();
  
  Logger.log("🏁 JS - Promedio de edad: " + promedio.toFixed(2));
  Logger.log("⏱️ JS - Tiempo invertido: " + (fin - inicio) + " milisegundos.");
}



/**
 * Fase 2: Transformación
 * Este archivo contiene la lógica matemática y de filtrado.
 */
function transformarDatos(matriz) {
  // 1. Separar la cabecera del cuerpo de datos
  const cabecera = matriz[0];
  const datos = matriz.slice(1);

  // 2. Identificar índices de columnas
  const idxChurn = cabecera.indexOf("Churn");
  const idxAge = cabecera.indexOf("Age");
  const idxContract = cabecera.indexOf("Contract_period");
  const idxFrequency = cabecera.indexOf("Avg_class_frequency_total");

  Logger.log("--- FASE DE TRANSFORMACIÓN INICIADA ---");

  // 3. FILTRAR: Solo usuarios que se dieron de baja (Churn == 1)
  const filtrados = datos.filter(fila => fila[idxChurn] == 1);

  // 4. MAPEAR: Seleccionar solo las columnas deseadas
  const mapeados = filtrados.map(fila => {
    return [
      fila[idxAge],      
      fila[idxContract],  
      // .toFixed(2) lo convierte a texto con 2 decimales
      Number(fila[idxFrequency]).toFixed(2) 
    ];
  });

  // 5. Añadir nuevas cabeceras para el reporte final
  // --- GUARDAMOS LA ESTRUCTURA ORIGINAL EN COMENTARIOS ---
  // const resultadoFinal = [
  //   ["Edad", "Meses_Contrato", "Frecuencia_Clases"],
  //   ...mapeados
  // ];
  
  // --- NUEVA ESTRUCTURA CLOUD: Guardamos el Total de Clientes de la nube en la cabecera para recuperarlo en el dashboard ---
  const resultadoFinal = [
    ["Edad", "Meses_Contrato", "Frecuencia_Clases", "Total_Clientes_Cloud"],
    ...mapeados.map((fila, index) => index === 0 ? [...fila, matriz.length - 1] : [...fila, ""])
  ];

  Logger.log("Filas después de filtrar (Churn=1): " + filtrados.length);
  Logger.log("--- FASE DE TRANSFORMACIÓN COMPLETADA ---");

  return resultadoFinal;
} 

function cargarDatos(datos) {
  // 1. Usamos nuestra función auxiliar para preparar la hoja
  const hojaDestino = obtenerOCrearHoja("Resultados_Churn");

  // 2. Limpiamos la hoja antes de volcar para evitar que queden datos residuales viejos
  hojaDestino.clearContents();

  // 3. Calculamos las dimensiones de nuestra matriz de datos
  const filas = datos.length;
  const columnas = datos[0].length;

  // 4. Volcamos los datos de golpe
  hojaDestino.getRange(1, 1, filas, columnas).setValues(datos);

  // Le damos un poco de estilo a la cabecera
  hojaDestino.getRange(1, 1, 1, columnas).setFontWeight("bold").setBackground("#d9ead3");
  
  Logger.log("--- FASE DE CARGA (L) COMPLETADA CON ÉXITO ---");
}

function calcularMetricasDashboard() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // 1. HOJA DE RESULTADOS (De aquí leeremos TODO ahora)
  const hojaResultados = ss.getSheetByName("Resultados_Churn");
  if (!hojaResultados) {
    console.error("Error: No se encontró la hoja 'Resultados_Churn'");
    return { total: 0, bajas: 0, porcentaje: "0%", color: "grey", mediaEdadBajas: 0, contratoMensual: 0, contratoAnual: 0 };
  }
  
  const datosResultados = hojaResultados.getDataRange().getValues();
  if (datosResultados.length <= 1) {
    return { total: 0, bajas: 0, porcentaje: "0%", color: "grey", mediaEdadBajas: 0, contratoMensual: 0, contratoAnual: 0 };
  }

  const cabeceraResultados = datosResultados[0];
  const totalBajas = datosResultados.length - 1;

  // --- CÓDIGO ANTIGUO COMENTADO (Buscaba en la pestaña estática antigua) ---
  /*
  const hojaOriginal = ss.getSheetByName("Datos_Crudos");
  if (!hojaOriginal) {
    console.error("Error: No se encontró la hoja 'Datos_Crudos'");
    return { total: 0, bajas: 0, porcentaje: "0%", color: "grey", mediaEdadBajas: 0, contratoMensual: 0, contratoAnual: 0 };
  }
  const datosCrudos = hojaOriginal.getDataRange().getValues();
  const cabecera = datosCrudos[0];
  const totalClientes = datosCrudos.length - 1;
  const idxEdad = cabecera.indexOf("Age");
  const idxContrato = cabecera.indexOf("Contract_period");
  const idxChurn = cabecera.indexOf("Churn");
  */

  // --- NUEVA LÓGICA CLOUD DIRECTA DESDE LA HOJA DE RESULTADOS ---
  const idxEdad = cabeceraResultados.indexOf("Edad");
  const idxContrato = cabeceraResultados.indexOf("Meses_Contrato");
  const idxTotalCloud = cabeceraResultados.indexOf("Total_Clientes_Cloud");

  // Recuperamos el total de clientes de la muestra de la nube que guardamos en la primera fila de datos
  const totalClientes = Number(datosResultados[1][idxTotalCloud]) || 150; 

  let sumaEdades = 0;
  let mensual = 0;
  let anual = 0;

  const filasDatos = datosResultados.slice(1);
  
  filasDatos.forEach(fila => {
    // Como esta hoja YA está filtrada con Churn = 1, procesamos todas las filas directamente
    sumaEdades += Number(fila[idxEdad]) || 0;
    const mesesContrato = Number(fila[idxContrato]);
    
    if (mesesContrato >= 12) {
      anual++;
    } else {
      mensual++;
    }
  });

  const mediaEdad = totalBajas > 0 ? (sumaEdades / totalBajas).toFixed(0) : 0;
  const porcentajeNum = totalClientes > 0 ? (totalBajas / totalClientes) * 100 : 0;
  const colorAlerta = porcentajeNum > 20 ? "#e74c3c" : "#2ecc71";

  return {
    total: totalClientes,
    bajas: totalBajas,
    porcentaje: porcentajeNum.toFixed(2) + "%",
    color: colorAlerta,
    mediaEdadBajas: mediaEdad,
    contratoMensual: mensual,
    contratoAnual: anual
  };
}



function obtenerOCrearHoja(nombreHoja) {
  const libro = SpreadsheetApp.getActiveSpreadsheet();
  let hoja = libro.getSheetByName(nombreHoja);
  
  // Si la hoja no existe, la creamos
  if (!hoja) {
    hoja = libro.insertSheet(nombreHoja);
    Logger.log("Hoja '" + nombreHoja + "' creada desde cero.");
  } else {
    // Si ya existe, la limpiamos por completo para evitar duplicados
    hoja.clear();
    Logger.log("Hoja '" + nombreHoja + "' encontrada y limpiada.");
  }
  
  return hoja;
}
